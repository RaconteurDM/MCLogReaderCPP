77 class CrtAllocator {
78 public:
79  static const bool kNeedFree = true;
80  void* Malloc(size_t size) {
81  if (size) // behavior of malloc(0) is implementation defined.
82  return RAPIDJSON_MALLOC(size);
83  else
84  return NULL; // standardize to returning NULL.
85  }
86  void* Realloc(void* originalPtr, size_t originalSize, size_t newSize) {
87  (void)originalSize;
88  if (newSize == 0) {
89  RAPIDJSON_FREE(originalPtr);
90  return NULL;
91  }
92  return RAPIDJSON_REALLOC(originalPtr, newSize);
93  }
94  static void Free(void *ptr) RAPIDJSON_NOEXCEPT { RAPIDJSON_FREE(ptr); }
95 
96  bool operator==(const CrtAllocator&) const RAPIDJSON_NOEXCEPT {
97  return true;
98  }
99  bool operator!=(const CrtAllocator&) const RAPIDJSON_NOEXCEPT {
100  return false;
101  }
102 };

123 template <typename BaseAllocator = CrtAllocator>
124 class MemoryPoolAllocator {
126 
128  struct ChunkHeader {
129  size_t capacity;
130  size_t size;
131  ChunkHeader *next;
132  };
133 
134  struct SharedData {
135  ChunkHeader *chunkHead;
136  BaseAllocator* ownBaseAllocator;
137  size_t refcount;
138  bool ownBuffer;
139  };
140 
141  static const size_t SIZEOF_SHARED_DATA = RAPIDJSON_ALIGN(sizeof(SharedData));
142  static const size_t SIZEOF_CHUNK_HEADER = RAPIDJSON_ALIGN(sizeof(ChunkHeader));
143 
144  static inline ChunkHeader *GetChunkHead(SharedData *shared)
145  {
146  return reinterpret_cast<ChunkHeader*>(reinterpret_cast<uint8_t*>(shared) + SIZEOF_SHARED_DATA);
147  }
148  static inline uint8_t *GetChunkBuffer(SharedData *shared)
149  {
150  return reinterpret_cast<uint8_t*>(shared->chunkHead) + SIZEOF_CHUNK_HEADER;
151  }
152 
153  static const size_t kDefaultChunkCapacity = RAPIDJSON_ALLOCATOR_DEFAULT_CHUNK_CAPACITY;
154 
155 public:
156  static const bool kNeedFree = false;
157 
159 
162  explicit
163  MemoryPoolAllocator(size_t chunkSize = kDefaultChunkCapacity, BaseAllocator* baseAllocator = 0) :
164  chunk_capacity_(chunkSize),
165  baseAllocator_(baseAllocator ? baseAllocator : RAPIDJSON_NEW(BaseAllocator)()),
166  shared_(static_cast<SharedData*>(baseAllocator_ ? baseAllocator_->Malloc(SIZEOF_SHARED_DATA + SIZEOF_CHUNK_HEADER) : 0))
167  {
168  RAPIDJSON_ASSERT(baseAllocator_ != 0);
169  RAPIDJSON_ASSERT(shared_ != 0);
170  if (baseAllocator) {
171  shared_->ownBaseAllocator = 0;
172  }
173  else {
174  shared_->ownBaseAllocator = baseAllocator_;
175  }
176  shared_->chunkHead = GetChunkHead(shared_);
177  shared_->chunkHead->capacity = 0;
178  shared_->chunkHead->size = 0;
179  shared_->chunkHead->next = 0;
180  shared_->ownBuffer = true;
181  shared_->refcount = 1;
182  }

194  MemoryPoolAllocator(void *buffer, size_t size, size_t chunkSize = kDefaultChunkCapacity, BaseAllocator* baseAllocator = 0) :
195  chunk_capacity_(chunkSize),
196  baseAllocator_(baseAllocator),
197  shared_(static_cast<SharedData*>(AlignBuffer(buffer, size)))
198  {
199  RAPIDJSON_ASSERT(size >= SIZEOF_SHARED_DATA + SIZEOF_CHUNK_HEADER);
200  shared_->chunkHead = GetChunkHead(shared_);
201  shared_->chunkHead->capacity = size - SIZEOF_SHARED_DATA - SIZEOF_CHUNK_HEADER;
202  shared_->chunkHead->size = 0;
203  shared_->chunkHead->next = 0;
204  shared_->ownBaseAllocator = 0;
205  shared_->ownBuffer = false;
206  shared_->refcount = 1;
207  }
208 
209  MemoryPoolAllocator(const MemoryPoolAllocator& rhs) RAPIDJSON_NOEXCEPT :
210  chunk_capacity_(rhs.chunk_capacity_),
211  baseAllocator_(rhs.baseAllocator_),
212  shared_(rhs.shared_)
213  {
214  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);

216  }
217  MemoryPoolAllocator& operator=(const MemoryPoolAllocator& rhs) RAPIDJSON_NOEXCEPT
218  {

220  ++rhs.shared_->refcount;
221 
222  this->~MemoryPoolAllocator();

227  }

232  ~MemoryPoolAllocator() RAPIDJSON_NOEXCEPT {
233  if (shared_->refcount > 1) {
234  --shared_->refcount;
235  return;
236  }
237  Clear();
238  BaseAllocator *a = shared_->ownBaseAllocator;
239  if (shared_->ownBuffer) {
240  baseAllocator_->Free(shared_);
241  }
242  RAPIDJSON_DELETE(a);
243  }

246  void Clear() RAPIDJSON_NOEXCEPT {
247  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
248  for (;;) {
249  ChunkHeader* c = shared_->chunkHead;
250  if (!c->next) {
251  break;
252  }
253  shared_->chunkHead = c->next;
254  baseAllocator_->Free(c);
255  }
256  shared_->chunkHead->size = 0;
257  }

262  size_t Capacity() const RAPIDJSON_NOEXCEPT {
263  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
264  size_t capacity = 0;
265  for (ChunkHeader* c = shared_->chunkHead; c != 0; c = c->next)
266  capacity += c->capacity;
267  return capacity;
268  }

274  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
275  size_t size = 0;
276  for (ChunkHeader* c = shared_->chunkHead; c != 0; c = c->next)
277  size += c->size;

284  bool Shared() const RAPIDJSON_NOEXCEPT {
285  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
286  return shared_->refcount > 1;
287  }

290  void* Malloc(size_t size) {
291  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
292  if (!size)
293  return NULL;
294 
295  size = RAPIDJSON_ALIGN(size);
296  if (RAPIDJSON_UNLIKELY(shared_->chunkHead->size + size > shared_->chunkHead->capacity))
297  if (!AddChunk(chunk_capacity_ > size ? chunk_capacity_ : size))
298  return NULL;
299 
300  void *buffer = GetChunkBuffer(shared_) + shared_->chunkHead->size;
301  shared_->chunkHead->size += size;
302  return buffer;
303  }
304 
306  void* Realloc(void* originalPtr, size_t originalSize, size_t newSize) {
307  if (originalPtr == 0)
308  return Malloc(newSize);
309 
310  RAPIDJSON_NOEXCEPT_ASSERT(shared_->refcount > 0);
311  if (newSize == 0)
312  return NULL;
313 
314  originalSize = RAPIDJSON_ALIGN(originalSize);
315  newSize = RAPIDJSON_ALIGN(newSize);
316 
317  // Do not shrink if new size is smaller than original
318  if (originalSize >= newSize)
319  return originalPtr;
320 
321  // Simply expand it if it is the last allocation and there is sufficient space
322  if (originalPtr == GetChunkBuffer(shared_) + shared_->chunkHead->size - originalSize) {
323  size_t increment = static_cast<size_t>(newSize - originalSize);
324  if (shared_->chunkHead->size + increment <= shared_->chunkHead->capacity) {
325  shared_->chunkHead->size += increment;
326  return originalPtr;
327  }
328  }
329 
330  // Realloc process: allocate and copy memory, do not free original buffer.
331  if (void* newBuffer = Malloc(newSize)) {
332  if (originalSize)
333  std::memcpy(newBuffer, originalPtr, originalSize);
334  return newBuffer;
335  }
336  else
337  return NULL;
338  }

359  bool AddChunk(size_t capacity) {
360  if (!baseAllocator_)
361  shared_->ownBaseAllocator = baseAllocator_ = RAPIDJSON_NEW(BaseAllocator)();
362  if (ChunkHeader* chunk = static_cast<ChunkHeader*>(baseAllocator_->Malloc(SIZEOF_CHUNK_HEADER + capacity))) {
363  chunk->capacity = capacity;
364  chunk->size = 0;
365  chunk->next = shared_->chunkHead;
366  shared_->chunkHead = chunk;
367  return true;
368  }
369  else
370  return false;
371  }

373  static inline void* AlignBuffer(void* buf, size_t &size)
374  {
375  RAPIDJSON_NOEXCEPT_ASSERT(buf != 0);
376  const uintptr_t mask = sizeof(void*) - 1;
377  const uintptr_t ubuf = reinterpret_cast<uintptr_t>(buf);
378  if (RAPIDJSON_UNLIKELY(ubuf & mask)) {
379  const uintptr_t abuf = (ubuf + mask) & ~mask;
380  RAPIDJSON_ASSERT(size >= abuf - ubuf);
381  buf = reinterpret_cast<void*>(abuf);
382  size -= abuf - ubuf;
383  }
384  return buf;
385  }

389  SharedData *shared_;
